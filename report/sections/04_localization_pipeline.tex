\subsection{Transformation Estimation}
This stage uses the inlier mask returned by RANSAC and discards all rejected correspondences. For the selected model (Similarity or Affine), the pipeline instantiates the appropriate linear system template defined in \autoref{subsec:transformation_models}, solves it with the least-squares method from \autoref{subsec:least_squares}, and converts the recovered parameter vector into a $3 \times 3$ matrix $M$. The output of this stage is only the estimated matrix $M$, which is consumed by the coordinate projection step.

\subsection{Coordinate Projection}
The estimated matrix $M$ is applied to the UAV-image center in homogeneous coordinates, where $W$ and $H$ are image width and height:
\begin{equation}
\mathbf{c}_{uav} =
\begin{bmatrix}
W/2 \\
H/2 \\
1
\end{bmatrix}
\end{equation}

\begin{equation}
\begin{bmatrix}
\tilde{X} \\
\tilde{Y} \\
1
\end{bmatrix}
= M\mathbf{c}_{uav}
\end{equation}

\begin{equation}
(X_{map}, Y_{map}) = (\tilde{X}, \tilde{Y}).
\end{equation}

\subsection{Complete Pipeline}

\begin{algorithm}[H]
\caption{UAV Localization Pipeline}
\begin{algorithmic}[1]
\REQUIRE UAV Image $I_{uav}$, Reference Map $I_{map}$
\ENSURE Global Coordinates $(X_{map}, Y_{map})$
\STATE Convert $I_{uav}$ and $I_{map}$ to grayscale.
\STATE Extract SIFT features and match using Lowe's ratio test.
\STATE \textbf{while} $i < \text{max\_iterations}$ \textbf{do}
\STATE \quad Sample minimal correspondence set and compute $M_{\text{temp}}$.
\STATE \quad Count inliers satisfying $d < \epsilon$.
\STATE \quad If current consensus is the largest, store $M_{\text{temp}}$ and inliers.
\STATE \textbf{end while}
\STATE Solve overdetermined system on full inlier set to obtain $M$.
\STATE Project UAV center $(x_c, y_c, 1)^\top$ through $M$ to get $(X_{map}, Y_{map})$.
\RETURN $(X_{map}, Y_{map})$
\end{algorithmic}
\end{algorithm}
